## 3. 后端编译与部署

### 3.1 后端编译

```bash
cd backend

# 安装依赖
go mod download

# 编译
go build -ldflags="-s -w" -o bin/computehub-api ./cmd/api

# 验证
./bin/computehub-api --version
```

### 3.2 后端Docker镜像

```dockerfile
# backend/deploy/Dockerfile
FROM golang:1.21-alpine AS builder

WORKDIR /app
COPY go.mod go.sum ./
RUN go mod download
COPY . .
RUN CGO_ENABLED=0 GOOS=linux go build -ldflags="-s -w" -o /app/api ./cmd/api

FROM alpine:3.19
RUN apk --no-cache add ca-certificates tzdata
COPY --from=builder /app/api /usr/local/bin/api
EXPOSE 8080
ENTRYPOINT ["api"]
```

### 3.3 后端配置

```yaml
# backend/config.yaml
server:
  port: 8080
  mode: release

database:
  host: mysql
  port: 3306
  user: api
  password: ${DB_PASSWORD}
  name: ai_zhaoshang
  max_connections: 100

redis:
  host: redis
  port: 6379
  password: ${REDIS_PASSWORD}
  db: 0

minio:
  endpoint: minio:9000
  access_key: ${MINIO_ACCESS_KEY}
  secret_key: ${MINIO_SECRET_KEY}
  bucket: content

ai:
  whisper_model: large-v3
  tts_engine: edge-tts
  offline_voice_model: para gan-small
```

## 7. 一键部署（Docker Compose）

```yaml
# deploy/docker-compose.yml
version: '3.8'

services:
  # 后端API
  api:
    build: ./backend/deploy
    ports:
      - "8080:8080"
    environment:
      - DB_PASSWORD=${DB_PASSWORD}
      - REDIS_PASSWORD=${REDIS_PASSWORD}
    depends_on:
      - mysql
      - redis

  # AI服务
  ai-service:
    build: ./ai-service/deploy
    ports:
      - "8000:8000"
    environment:
      - API_ENDPOINT=http://api:8080
    depends_on:
      - api

  # MySQL
  mysql:
    image: mysql:8.0
    environment:
      MYSQL_ROOT_PASSWORD: ${DB_PASSWORD}
      MYSQL_DATABASE: ai_zhaoshang
    volumes:
      - mysql-data:/var/lib/mysql
      - ./database/init.sql:/docker-entrypoint-initdb.d/init.sql

  # Redis
  redis:
    image: redis:7-alpine
    command: redis-server --requirepass ${REDIS_PASSWORD}
    volumes:
      - redis-data:/data

  # Nginx
  nginx:
    image: nginx:alpine
    ports:
      - "80:80"
      - "443:443"
    volumes:
      - ./nginx/conf.d:/etc/nginx/conf.d
      - ./frontend/dist:/usr/share/nginx/html
    depends_on:
      - api

  # MinIO
  minio:
    image: minio/minio:latest
    command: server /data --console-address ":9001"
    ports:
      - "9000:9000"
      - "9001:9001"
    environment:
      MINIO_ROOT_USER: ${MINIO_ACCESS_KEY}
      MINIO_ROOT_PASSWORD: ${MINIO_SECRET_KEY}
    volumes:
      - minio-data:/data

volumes:
  mysql-data:
  redis-data:
  minio-data:
```

### 7.1 启动

```bash
# 创建.env文件

# 编辑.env填入实际值

# 一键启动
docker compose up -d

# 查看状态
docker compose ps

# 查看日志
docker compose logs -f api
```
