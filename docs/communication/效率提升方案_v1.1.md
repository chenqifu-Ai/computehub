### 1.3 WSHub 核心

```go
// WSHub WebSocket 管理中心
type WSHub struct {
    mu      sync.RWMutex
    clients map[string]*WSClient // nodeID → WSClient
    metrics CommMetrics          // 通信指标（Phase 4 扩展用）
}

// NewWSHub 创建 WSHub 实例
func NewWSHub() *WSHub {
    return &WSHub{
        clients: make(map[string]*WSClient),
    }
}

// Register 注册 WS 连接
func (h *WSHub) Register(nodeID, platform string, conn *websocket.Conn) *WSClient {
    h.mu.Lock()
    defer h.mu.Unlock()

    // 如果已有旧连接，先关掉
    if old, ok := h.clients[nodeID]; ok {
        old.Cancel()
        old.Conn.Close()
    }

    ctx, cancel := context.WithCancel(context.Background())
    client := &WSClient{
        NodeID:   nodeID,
        Platform: platform,
        Conn:     conn,
        Topics:   map[string]bool{"general": true}, // 默认订阅 general
        RegAt:    time.Now(),
        Ctx:      ctx,
        Cancel:   cancel,
    }
    h.clients[nodeID] = client

    h.metrics.WsConnections = len(h.clients)
    log.Printf("📡 WS Hub: %s 已连接 (共 %d 个连接)", nodeID, len(h.clients))
    return client
}

// Unregister 注销 WS 连接
func (h *WSHub) Unregister(nodeID string) {
    h.mu.Lock()
    defer h.mu.Unlock()

    if client, ok := h.clients[nodeID]; ok {
        client.Cancel()
        client.Conn.Close()
        delete(h.clients, nodeID)
        h.metrics.WsConnections = len(h.clients)
        log.Printf("📡 WS Hub: %s 已断开 (剩余 %d 个连接)", nodeID, len(h.clients))
    }
}

// Subscribe 设置节点订阅的话题
func (h *WSHub) Subscribe(nodeID string, topics []string) {
    h.mu.Lock()
    defer h.mu.Unlock()

    if client, ok := h.clients[nodeID]; ok {
        client.Topics = make(map[string]bool)
        for _, t := range topics {
            client.Topics[t] = true
        }
        log.Printf("📡 WS Hub: %s 订阅话题 %v", nodeID, topics)
    }
}

// GetClient 获取指定节点连接
func (h *WSHub) GetClient(nodeID string) *WSClient {
    h.mu.RLock()
    defer h.mu.RUnlock()
    return h.clients[nodeID]
}

// OnlineCount 返回在线 WS 连接数
func (h *WSHub) OnlineCount() int {
    h.mu.RLock()
    defer h.mu.RUnlock()
    return len(h.clients)
}

// ListClients 列出所有在线 WS 客户端
func (h *WSHub) ListClients() []string {
    h.mu.RLock()
    defer h.mu.RUnlock()
    ids := make([]string, 0, len(h.clients))
    for id := range h.clients {
        ids = append(ids, id)
    }
    return ids
}
```

### 1.4 FanOut — 核心分发方法

```go
// FanOut 向指定话题的所有订阅者分发消息
// - topic: 话题名称
// - msg: 消息体（会自动添加 sender, timestamp, seq）
// - exceptID: 排除的 nodeID（"", "gateway" 表示不排除任何节点）
// 返回成功投递数
func (h *WSHub) FanOut(topic string, msg interface{}, exceptID string) int {
    h.mu.RLock()
    defer h.mu.RUnlock()

    payload, err := json.Marshal(msg)
    if err != nil {
        log.Printf("📡 WS FanOut 序列化失败: %v", err)
        return 0
    }

    delivered := 0
    for nodeID, client := range h.clients {
        // 跳过排除的节点
        if nodeID == exceptID || exceptID == "" {
            continue
        }
        // 跳过未订阅该话题的节点
        if !client.Topics[topic] {
            continue
        }

        // 写入 WS（非阻塞，超时 2s）
        err := client.Conn.SetWriteDeadline(time.Now().Add(2 * time.Second))
        if err != nil {
            log.Printf("📡 WS FanOut 设置 deadline 失败 (%s): %v", nodeID, err)
            continue
        }

        err = client.Conn.WriteMessage(websocket.TextMessage, payload)
        if err != nil {
            log.Printf("📡 WS FanOut 投递失败 (%s): %v", nodeID, err)
            // 不立即断开，让健康循环处理
            continue
        }
        delivered++
    }

    h.metrics.MessagesSent += int64(delivered)
    return delivered
}

// FanOutAll 向所有在线节点分发（跳过 exceptID）
func (h *WSHub) FanOutAll(msg interface{}, exceptID string) int {
    h.mu.RLock()
    defer h.mu.RUnlock()

    payload, err := json.Marshal(msg)
    if err != nil {
        log.Printf("📡 WS FanOut 序列化失败: %v", err)
        return 0
    }

    delivered := 0
    for nodeID, client := range h.clients {
        if nodeID == exceptID {
            continue
        }

        err := client.Conn.SetWriteDeadline(time.Now().Add(2 * time.Second))
        if err != nil {
            continue
        }
        err = client.Conn.WriteMessage(websocket.TextMessage, payload)
        if err == nil {
            delivered++
        }
    }

    h.metrics.MessagesSent += int64(delivered)
    return delivered
}
```

### 1.5 WS Upgrade HTTP Handler

```go
// HandleWSUpgrade 处理 HTTP→WS 升级
// 客户端连接后在 TextMessage 中发送注册信息:
//   {"action": "register", "node_id": "worker-arm", "platform": "android/arm64", "subscribe": ["general", "alerts"]}
func (h *WSHub) HandleWSUpgrade(w http.ResponseWriter, r *http.Request) {
    conn, err := upgrader.Upgrade(w, r, nil)
    if err != nil {
        log.Printf("📡 WS upgrade 失败: %v", err)
        return
    }

    var nodeID string
    var platform string

    // 读取第一条消息：注册信息
    _, msg, err := conn.ReadMessage()
    if err != nil {
        log.Printf("📡 WS 读注册消息失败: %v", err)
        conn.Close()
        return
    }

    var regMsg struct {
        Action    string   `json:"action"`
        NodeID    string   `json:"node_id"`
        Platform  string   `json:"platform"`
        Subscribe []string `json:"subscribe,omitempty"`
    }
    if err := json.Unmarshal(msg, &regMsg); err != nil || regMsg.Action != "register" || regMsg.NodeID == "" {
        log.Printf("📡 WS 非法注册消息: %s", string(msg))
        conn.WriteJSON(map[string]string{"error": "invalid register"})
        conn.Close()
        return
    }

    nodeID = regMsg.NodeID
    platform = regMsg.Platform

    // 注册客户端
    client := h.Register(nodeID, platform, conn)

    // 设置订阅
    if len(regMsg.Subscribe) > 0 {
        h.Subscribe(nodeID, regMsg.Subscribe)
    }

    // 回复注册成功
    conn.WriteJSON(map[string]interface{}{
        "status":  "ok",
        "node_id": nodeID,
        "message": "registered",
        "seq":     client.LastSeq,
    })

    // 保持连接，处理后续消息
    go h.readPump(client)
    go h.writePump(client)
}

// readPump 持续读取客户端消息（心跳、订阅变更、内容消息）
func (h *WSHub) readPump(client *WSClient) {
    defer func() {
        h.Unregister(client.NodeID)
    }()

    for {
        select {
        case <-client.Ctx.Done():
            return
        default:
        }

        _, msg, err := client.Conn.ReadMessage()
        if err != nil {
            // 正常断开或超时
            if websocket.IsUnexpectedCloseError(err, websocket.CloseNormalClosure, websocket.CloseGoingAway) {
                log.Printf("📡 WS readPump (%s): %v", client.NodeID, err)
            }
            return
        }

        // 解析消息类型
        var envelope struct {
            Action  string          `json:"action"`
            Topic   string          `json:"topic,omitempty"`
            Payload json.RawMessage `json:"payload,omitempty"`
        }
        if err := json.Unmarshal(msg, &envelope); err != nil {
            continue
        }

        switch envelope.Action {
        case "ping":
            client.Conn.WriteJSON(map[string]string{"action": "pong"})

        case "subscribe":
            var subReq struct {
                Topics []string `json:"topics"`
            }
            json.Unmarshal(envelope.Payload, &subReq)
            h.Subscribe(client.NodeID, subReq.Topics)
            client.Conn.WriteJSON(map[string]string{"action": "subscribed"})

        case "hall_send":
            // 消息转发到 Hall（走现有 PostHallMessage 逻辑）
            hall.SendMessage(envelope.Topic, client.NodeID, "", string(envelope.Payload))

        case "broadcast":
            // 消息广播给所有其他节点
            h.FanOutAll(envelope.Payload, client.NodeID)
        }
    }
}

// writePump 定期向客户端发送心跳
func (h *WSHub) writePump(client *WSClient) {
    ticker := time.NewTicker(10 * time.Second)
    defer ticker.Stop()

    for {
        select {
        case <-client.Ctx.Done():
            return
        case <-ticker.C:
            err := client.Conn.WriteJSON(map[string]string{"action": "ping"})
            if err != nil {
                log.Printf("📡 WS writePump (%s): %v", client.NodeID, err)
                return
            }
        }
    }
}

// Close 关闭所有 WS 连接
func (h *WSHub) Close() {
    h.mu.Lock()
    defer h.mu.Unlock()

    for _, client := range h.clients {
        client.Cancel()
        client.Conn.Close()
    }
    h.clients = make(map[string]*WSClient)
    log.Printf("📡 WS Hub: 已关闭所有连接")
}
```

```go
// StartWSLoop 启动 WebSocket 连接循环（替代旧 StartPollLoop）
// 自动连接 → 注册 → 消息接收 → 断线重连
func (hc *HallClient) StartWSLoop() {
    go func() {
        time.Sleep(3 * time.Second) // 等 Gateway 就绪

        // 无限重连循环
        backoff := 1 * time.Second
        maxBackoff := 30 * time.Second

        for {
            select {
            case <-hc.stopCh:
                fmt.Printf(" [HallClient:%s] ⏹ 停止 WS 循环\n", hc.nodeID)
                return
            default:
            }

            // 尝试 WS 连接
            err := hc.connectWS()
            if err != nil {
                fmt.Printf(" [HallClient:%s] ⚠️ WS 连接失败: %v (重试 %v)...\n", hc.nodeID, err, backoff)
                
                select {
                case <-hc.stopCh:
                    return
                case <-time.After(backoff):
                }

                // 指数退避
                backoff *= 2
                if backoff > maxBackoff {
                    backoff = maxBackoff
                }
                continue
            }

            // 连接成功，重置退避
            backoff = 1 * time.Second

            // 注册到 WS Hub
            err = hc.registerOnWS()
            if err != nil {
                fmt.Printf(" [HallClient:%s] ⚠️ WS 注册失败: %v\n", hc.nodeID, err)
                hc.wsConn.Close()
                continue
            }

            // 发自我介绍
            hc.postMessage("general", hc.nodeID, hc.nodeName, "all",
                fmt.Sprintf("🏛️ %s (%s) 已加入大厅 (WS)", hc.nodeName, hc.nodeID))

            // 进入消息接收循环
            hc.wsReadLoop()
        }
    }()
}

// connectWS 建立 WebSocket 连接
func (hc *HallClient) connectWS() error {
    wsURL := strings.Replace(hc.gatewayURL, "http://", "ws://", 1) + "/api/v1/ws"
    
    dialer := websocket.Dialer{
        HandshakeTimeout: 5 * time.Second,
    }
    
    conn, _, err := dialer.Dial(wsURL, nil)
    if err != nil {
        return fmt.Errorf("dial WS %s: %w", wsURL, err)
    }

    hc.wsMu.Lock()
    hc.wsConn = conn
    hc.wsMu.Unlock()

    fmt.Printf(" [HallClient:%s] ✅ WS 已连接: %s\n", hc.nodeID, wsURL)
    return nil
}

// registerOnWS 通过 WS 发送注册信息
func (hc *HallClient) registerOnWS() error {
    regMsg := map[string]interface{}{
        "action":    "register",
        "node_id":   hc.nodeID,
        "platform":  fmt.Sprintf("%s/%s", runtime.GOOS, runtime.GOARCH),
        "subscribe": []string{"general", "alerts", hc.nodeID},
    }

    hc.wsMu.Lock()
    err := hc.wsConn.WriteJSON(regMsg)
    hc.wsMu.Unlock()
    if err != nil {
        return fmt.Errorf("write register: %w", err)
    }

    // 等待注册确认
```

```go
// wsReadLoop 持续读取 WS 消息（替代旧 pollOnce）
func (hc *HallClient) wsReadLoop() {
    for {
        select {
        case <-hc.stopCh:
            return
        default:
        }

        _, msg, err := hc.wsConn.ReadMessage()
        if err != nil {
            fmt.Printf(" [HallClient:%s] ⚠️ WS 读消息失败: %v (触发重连)\n", hc.nodeID, err)
            return // 触发 StartWSLoop 重连
        }

        // 解析消息
        var envelope struct {
            Action  string          `json:"action"`
            Type    string          `json:"type,omitempty"`
            Topic   string          `json:"topic,omitempty"`
            Payload json.RawMessage `json:"payload,omitempty"`
            Seq     int64           `json:"seq,omitempty"`
            From    string          `json:"from,omitempty"`
            FromName string         `json:"from_name,omitempty"`
            Content string          `json:"content,omitempty"`
        }
        json.Unmarshal(msg, &envelope)

        // 处理不同类型消息
        switch {
        case envelope.Action == "ping":
            hc.wsMu.Lock()
            hc.wsConn.WriteJSON(map[string]string{"action": "pong"})
            hc.wsMu.Unlock()

        case envelope.Type == "hall_message" || envelope.Topic != "":
            // 构造 HallMessage 并触发 Agent
            hallMsg := HallMessage{
                MsgID:     fmt.Sprintf("ws-%d", time.Now().UnixNano()),
                Topic:     envelope.Topic,
                From:      envelope.From,
                FromName:  envelope.FromName,
                To:        "all",
                Content:   envelope.Content,
                Timestamp: time.Now(),
                Seq:       envelope.Seq,
            }

            // 去重
            hc.mu.Lock()
            if hc.processedIDs[hallMsg.MsgID] {
                hc.mu.Unlock()
                continue
            }
            hc.processedIDs[hallMsg.MsgID] = true
            if hallMsg.Seq > hc.lastSeqs[hallMsg.Topic] {
                hc.lastSeqs[hallMsg.Topic] = hallMsg.Seq
            }
            hc.mu.Unlock()

            // 跳过自己发的消息
            if hallMsg.From == hc.nodeID {
                continue
            }

            // 触发 Agent 回复
            go hc.thinkAndReply(hallMsg)

        case envelope.Type == "arc_net_broadcast":
            // ARC-NET 广播消息（不调 LLM，直接处理拓扑）
            // 现有 handleArcNetBroadcast 逻辑仍保留处理 HTTP 版本
            // WS 版本的消息直接透传
            // TODO: 跨节点 P2P 消息（Phase 3）

        default:
            // 未知消息，忽略
        }
    }
```

```go
// postMessage 发送消息 — WS 优先，HTTP fallback
func (hc *HallClient) postMessage(topic, from, fromName, to, content string) {
    // 构造消息
    msg := map[string]interface{}{
        "action": "hall_send",
        "topic":  topic,
        "payload": map[string]interface{}{
            "from":       from,
            "node_name":  fromName,
            "to":         to,
            "content":    content,
        },
    }

    // WS 优先发送
    hc.wsMu.Lock()
    if hc.wsConn != nil {
        err := hc.wsConn.WriteJSON(msg)
        hc.wsMu.Unlock()
        if err == nil {
            return // WS 发送成功
        }
        // WS 失败，走 HTTP fallback
        fmt.Printf(" [HallClient:%s] ⚠️ WS 发送失败, fallback HTTP: %v\n", hc.nodeID, err)
    } else {
        hc.wsMu.Unlock()
    }

    // HTTP fallback（保留旧方式）
    payload := map[string]string{
        "topic":     topic,
        "from":      from,
        "node_name": fromName,
        "to":        to,
        "content":   content,
    }
    body, _ := json.Marshal(payload)
    resp, err := hc.httpClient.Post(
        hc.gatewayURL+"/api/v1/hall/post",
        "application/json",
        bytes.NewReader(body),
    )
    if err != nil {
        fmt.Printf(" [HallClient:%s] ❌ HTTP 发消息失败: %v\n", hc.nodeID, err)
        return
    }
    resp.Body.Close()
}
```
